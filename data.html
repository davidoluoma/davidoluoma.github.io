-- Customer support ticket analysis — Ratedby10 (300+ daily records)
SELECT 
  issue_category,
  COUNT(*) AS ticket_count,
  AVG(resolution_time_hrs) AS avg_resolution,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 1) AS pct_share
FROM support_tickets
WHERE created_date BETWEEN '2024-01-01' AND '2024-04-30'
GROUP BY issue_category
ORDER BY ticket_count DESC;
